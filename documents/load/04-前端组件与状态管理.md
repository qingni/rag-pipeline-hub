## 2. 核心组件详解

### 2.1 DocumentUploader.vue (文档上传器)

**功能**:
1. 支持点击上传和拖拽上传
2. 文件格式和大小验证
3. 上传进度实时显示
4. 上传成功/失败提示

**关键特性**:
- 📤 拖拽上传: 用户友好的交互方式
- ✅ 格式验证: 支持PDF/DOC/DOCX/TXT/MD
- 📊 进度反馈: 实时显示上传进度
- 🔒 重复检测: 避免重复上传同名文件

**核心代码**:
```vue
<template>
  <div class="document-uploader">
    <t-upload
      v-model="files"
      :auto-upload="false"
      :draggable="true"
      :accept="acceptedFormats"
      :max="1"
      :size-limit="{ size: MAX_FILE_SIZE, unit: 'B' }"
      :before-upload="handleBeforeUpload"
      @change="handleFileChange"
      theme="file"
    >
      <template #drag-content>
        <div class="upload-area">
          <UploadCloudIcon :size="32" />
          <p>点击上传 / 拖拽到此区域</p>
        </div>
      </template>
    </t-upload>
    
    <!-- 进度条 -->
    <div v-if="uploading">
      <t-progress :percentage="progress" :status="uploadStatus" />
    </div>
    
    <!-- 提示信息 -->
    <t-alert v-if="error" theme="error" :message="error" />
    <t-alert v-if="uploadSuccess" theme="success" message="上传成功" />
  </div>
</template>

<script setup>
const MAX_FILE_SIZE = 52428800 // 50MB
const acceptedFormats = '.pdf,.doc,.docx,.txt,.md,.markdown'

function validateFile(file) {
  const allowedExtensions = ['.pdf', '.doc', '.docx', '.txt', '.md', '.markdown']
  const fileExt = '.' + file.name.split('.').pop().toLowerCase()
  
  if (!allowedExtensions.includes(fileExt)) {
    throw new Error(`不支持的文件格式: ${fileExt}`)
  }
  
  if (file.size > MAX_FILE_SIZE) {
    throw new Error(`文件大小超过限制 (最大50MB)`)
  }
  
  if (file.size === 0) {
    throw new Error('文件为空')
  }
  
  // 检查重复文件名
  const existingDoc = documentStore.documents.find(
    doc => doc.filename === file.name
  )
  if (existingDoc) {
    throw new Error(`文件"${file.name}"已存在`)
  }
  
  return true
}

async function handleFileChange(fileList) {
  if (fileList.length === 0) return
  
  const file = fileList[0].raw
  
  try {
    uploading.value = true
    const document = await documentStore.uploadDocument(file)
    uploadSuccess.value = true
    emit('upload-complete', document)
  } catch (err) {
    error.value = err.message
  } finally {
    uploading.value = false
  }
}
</script>
```

**数据流**:
```
用户选择文件
  │
  ▼
validateFile() → 验证格式/大小/重复
  │
  ▼
handleFileChange()
  │
  ▼
documentStore.uploadDocument()
  │
  ▼
POST /api/v1/documents
  │
  ▼
emit('upload-complete', document)
```

## 3. 状态管理

### 3.1 Pinia Store 结构

**documentStore** (文档管理):
```javascript
export const useDocumentStore = defineStore('document', () => {
  // State
  const documents = ref([])
  const selectedDocument = ref(null)
  const loading = ref(false)
  const uploadProgress = ref(0)
  const currentPage = ref(1)
  const pageSize = ref(20)
  const totalCount = ref(0)
  const filters = ref({
    search: '',
    format: null,
    status: null
  })
  
  // Actions
  async function uploadDocument(file) {
    const formData = new FormData()
    formData.append('file', file)
    
    const response = await axios.post('/api/v1/documents', formData, {
      onUploadProgress: (progressEvent) => {
        uploadProgress.value = Math.round(
          (progressEvent.loaded * 100) / progressEvent.total
        )
      }
    })
    
    if (response.data.success) {
      // 刷新列表
      await fetchDocuments(currentPage.value)
      return response.data.data
    }
  }
  
  async function fetchDocuments(page = 1, newFilters = {}) {
    loading.value = true
    try {
      const params = {
        page,
        page_size: pageSize.value,
        ...filters.value,
        ...newFilters
      }
      
      const response = await axios.get('/api/v1/documents', { params })
      
      if (response.data.success) {
        documents.value = response.data.data.items
        totalCount.value = response.data.data.total
        currentPage.value = page
      }
    } finally {
      loading.value = false
    }
  }
  
  async function deleteDocument(documentId) {
    await axios.delete(`/api/v1/documents/${documentId}`)
    await fetchDocuments(currentPage.value)
  }
  
  function selectDocument(document) {
    selectedDocument.value = document
  }
  
  return {
    documents,
    selectedDocument,
    loading,
    uploadProgress,
    currentPage,
    pageSize,
    totalCount,
    filters,
    uploadDocument,
    fetchDocuments,
    deleteDocument,
    selectDocument
  }
})
```

**processingStore** (处理管理):
```javascript
export const useProcessingStore = defineStore('processing', () => {
  // State
  const currentResult = ref(null)
  const status = ref('idle')
  const loading = ref(false)
  const error = ref(null)
  
  // Actions
  async function loadDocument(documentId, loaderType = null) {
    loading.value = true
    status.value = 'processing'
    error.value = null
    
    try {
      const payload = { document_id: documentId }
      if (loaderType) {
        payload.loader_type = loaderType
      }
      
      const response = await axios.post('/api/v1/processing/load', payload)
      
      if (response.data.success) {
        // 获取完整结果
        const result = await fetchResultById(response.data.data.id)
        currentResult.value = result
        status.value = 'completed'
        return result
      }
    } catch (err) {
      error.value = err.response?.data?.message || err.message
      status.value = 'failed'
      throw err
    } finally {
      loading.value = false
    }
  }
  
  async function fetchResultById(resultId) {
    const response = await axios.get(`/api/v1/processing/results/detail/${resultId}`)
    return response.data.data
  }
  
  function reset() {
    currentResult.value = null
    status.value = 'idle'
    error.value = null
  }
  
  return {
    currentResult,
    status,
    loading,
    error,
    loadDocument,
    fetchResultById,
    reset
  }
})
```

## 4. 工作流程

### 4.1 完整加载流程

```
1. 用户上传文档
   └─> DocumentUploader.vue
       └─> documentStore.uploadDocument()
           └─> POST /api/v1/documents
               └─> 文件保存到服务器
               └─> 创建Document记录
               └─> 刷新文档列表

2. 用户选择文档
   └─> DocumentList.vue → handleSelect()
       └─> documentStore.selectDocument(doc)
           └─> 更新selectedDocument
           └─> 切换到"文档预览"Tab

3. 用户选择加载器(可选)
   └─> LoaderSelector.vue
       └─> 更新loaderType

4. 用户点击"开始加载"
   └─> DocumentLoad.vue → handleLoad()
       └─> processingStore.loadDocument(docId, loaderType)
           └─> POST /api/v1/processing/load
               └─> 后端执行加载
               └─> 返回result_id
           └─> GET /api/v1/processing/results/detail/{result_id}
               └─> 获取完整结果数据
           └─> 更新currentResult
           └─> 自动切换到"加载结果"Tab

5. 用户查看结果
   └─> ResultPreview.vue
       ├─> 显示元数据
       ├─> 显示文本内容(分页)
       └─> 提供下载功能
```

### 4.2 状态流转

```
初始状态 (idle)
  │
  ▼
用户上传文档
  ├─ documentStore.uploadDocument()
  └─ 文件上传中
      │
      ▼
  上传成功
      ├─ 刷新文档列表
      └─ emit('upload-complete', document)
  │
  ▼
用户选择文档
  ├─ documentStore.selectDocument(doc)
  └─ selectedDocument更新
  │
  ▼
用户点击加载
  ├─ processingStore.loadDocument()
  ├─ status: 'processing'
  └─ 调用API
      │
      ▼
  加载完成
      ├─ status: 'completed'
      ├─ currentResult更新
      └─> 展示结果
```

### 4.3 错误处理流程

```
API调用失败
  │
```
