## 1. 数据模型概述

### 1.1 存储架构

搜索查询模块主要存储搜索历史和配置信息：

| 存储类型 | 用途 | 数据内容 |
|---------|------|----------|
| 关系数据库 | 历史记录 | 搜索历史、配置快照 |
| 内存缓存 | 临时数据 | 搜索结果缓存（可选） |

### 1.2 设计原则

1. **轻量存储**: 只存储必要的历史信息
2. **快速查询**: 支持按时间排序的快速查询
3. **自动清理**: 自动清理过期历史记录
4. **配置快照**: 保存搜索时的配置信息

## 3. 数据库模型详解

### 3.1 SearchHistory 模型

```python
# backend/src/models/search.py

from sqlalchemy import Column, String, Integer, Float, DateTime, JSON, Text
from sqlalchemy.orm import relationship
from datetime import datetime
import hashlib

class SearchHistory(Base):
    """搜索历史模型"""
    __tablename__ = "search_history"
    
    # 主键
    id = Column(String(36), primary_key=True)
    
    # 查询信息
    query = Column(Text, nullable=False, comment="查询文本")
    query_hash = Column(String(64), nullable=False, index=True, comment="查询文本哈希")
    
    # 搜索配置
    index_ids = Column(JSON, default=list, comment="搜索的索引ID列表")
    top_k = Column(Integer, nullable=False, comment="请求返回数量")
    threshold = Column(Float, nullable=False, comment="相似度阈值")
    metric_type = Column(String(20), nullable=True, comment="度量类型")
    
    # 结果信息
    result_count = Column(Integer, default=0, comment="实际返回结果数")
    search_time_ms = Column(Float, nullable=False, comment="搜索耗时(毫秒)")
    results_summary = Column(JSON, nullable=True, comment="结果摘要(前几条)")
    
    # 时间戳
    created_at = Column(DateTime, default=datetime.utcnow, index=True, comment="创建时间")
    
    def __repr__(self):
        return f"<SearchHistory(id={self.id}, query={self.query[:20]}...)>"
    
    @staticmethod
    def compute_query_hash(query: str) -> str:
        """计算查询文本的哈希值"""
        return hashlib.sha256(query.encode()).hexdigest()
    
    @property
    def config(self) -> dict:
        """获取搜索配置"""
        return {
            "index_ids": self.index_ids or [],
            "top_k": self.top_k,
            "threshold": self.threshold,
            "metric_type": self.metric_type
        }
```

### 3.2 字段说明

| 字段 | 类型 | 说明 | 约束 |
|------|------|------|------|
| id | String(36) | 主键UUID | PK |
| query | Text | 查询文本 | NOT NULL |
| query_hash | String(64) | 查询哈希 | NOT NULL, INDEX |
| index_ids | JSON | 索引ID列表 | DEFAULT [] |
| top_k | Integer | 请求数量 | NOT NULL |
| threshold | Float | 相似度阈值 | NOT NULL |
| metric_type | String(20) | 度量类型 | NULLABLE |
| result_count | Integer | 结果数量 | DEFAULT 0 |
| search_time_ms | Float | 搜索耗时 | NOT NULL |
| results_summary | JSON | 结果摘要 | NULLABLE |
| created_at | DateTime | 创建时间 | DEFAULT NOW, INDEX |

## 5. 历史记录管理

### 5.1 保存历史记录

```python
class SearchService:
    MAX_HISTORY_COUNT = 50  # 最大历史记录数
    MAX_RESULTS_SUMMARY = 5  # 摘要中保存的结果数
    
    async def save_history(
        self,
        query: str,
        config: dict,
        results: list,
        search_time_ms: float
    ) -> SearchHistory:
        """保存搜索历史"""
        # 创建结果摘要
        results_summary = [
            {
                "id": r["id"],
                "score": r["score"],
                "text_preview": r["text"][:100] + "..." if len(r["text"]) > 100 else r["text"],
                "index_name": r["index_name"]
            }
            for r in results[:self.MAX_RESULTS_SUMMARY]
        ]
        
        # 创建历史记录
        history = SearchHistory(
            id=generate_uuid(),
            query=query,
            query_hash=SearchHistory.compute_query_hash(query),
            index_ids=config.get("index_ids", []),
            top_k=config.get("top_k", 10),
            threshold=config.get("threshold", 0.5),
            metric_type=config.get("metric_type"),
            result_count=len(results),
            search_time_ms=search_time_ms,
            results_summary=results_summary
        )
        
        self.db.add(history)
        await self.db.commit()
        
        # 清理超出限制的旧记录
        await self._cleanup_old_history()
        
        return history
    
    async def _cleanup_old_history(self):
        """清理超出限制的旧历史记录"""
        # 获取当前记录数
        count = await self.db.query(func.count(SearchHistory.id)).scalar()
        
        if count > self.MAX_HISTORY_COUNT:
            # 删除最旧的记录
            excess = count - self.MAX_HISTORY_COUNT
            oldest = await self.db.query(SearchHistory)\
                .order_by(SearchHistory.created_at.asc())\
                .limit(excess)\
                .all()
            
            for record in oldest:
                await self.db.delete(record)
            
            await self.db.commit()
```

### 5.2 获取历史记录

```python
async def get_history(
    self,
    page: int = 1,
    page_size: int = 20,
    limit: int = 50
) -> dict:
    """获取搜索历史列表"""
    # 计算实际限制
    actual_limit = min(page_size, limit)
    offset = (page - 1) * page_size
    
    # 查询历史记录
    query = self.db.query(SearchHistory)\
        .order_by(SearchHistory.created_at.desc())\
        .offset(offset)\
        .limit(actual_limit)
    
    items = await query.all()
    
    # 获取总数
    total = await self.db.query(func.count(SearchHistory.id)).scalar()
    
    return {
        "items": [
            {
                "id": h.id,
                "query": h.query,
                "result_count": h.result_count,
                "search_time_ms": h.search_time_ms,
                "config": h.config,
                "created_at": h.created_at.isoformat()
            }
            for h in items
        ],
        "total": min(total, limit),
        "page": page,
        "page_size": page_size
    }
```

### 5.3 删除历史记录

```python
async def delete_history(self, history_id: str) -> bool:
    """删除单条历史记录"""
    history = await self.db.get(SearchHistory, history_id)
    if not history:
        return False
    
    await self.db.delete(history)
    await self.db.commit()
    return True

async def clear_history(self) -> int:
    """清空所有历史记录"""
    result = await self.db.execute(
        delete(SearchHistory)
    )
    await self.db.commit()
    return result.rowcount
```

## 6. 数据清理策略

### 6.1 自动清理规则

| 规则 | 条件 | 动作 |
|------|------|------|
| 数量限制 | 超过50条 | 删除最旧的记录 |
| 时间限制 | 超过30天 | 可选：删除过期记录 |

### 6.2 定时清理任务

```python
from datetime import datetime, timedelta

async def cleanup_expired_history(days: int = 30):
    """清理过期的历史记录"""
    cutoff_date = datetime.utcnow() - timedelta(days=days)
    
    result = await db.execute(
        delete(SearchHistory).where(
            SearchHistory.created_at < cutoff_date
        )
    )
    
    await db.commit()
    
    return result.rowcount
```

### 6.3 手动清理接口

```python
# API 端点
@router.delete("/search/history/cleanup")
async def cleanup_history(
    days: int = Query(default=30, ge=1, le=365),
    db: Session = Depends(get_db)
):
    """清理指定天数前的历史记录"""
    service = SearchService(db)
    deleted_count = await service.cleanup_expired_history(days)
    
    return {
        "deleted_count": deleted_count,
        "cleanup_before": (datetime.utcnow() - timedelta(days=days)).isoformat()
    }
```

## 8. 数据统计

### 8.1 搜索统计查询

```python
async def get_search_statistics(self, days: int = 7) -> dict:
    """获取搜索统计信息"""
    cutoff_date = datetime.utcnow() - timedelta(days=days)
    
    # 总搜索次数
    total_searches = await self.db.query(
        func.count(SearchHistory.id)
    ).filter(
        SearchHistory.created_at >= cutoff_date
    ).scalar()
    
    # 平均搜索耗时
    avg_search_time = await self.db.query(
        func.avg(SearchHistory.search_time_ms)
    ).filter(
        SearchHistory.created_at >= cutoff_date
    ).scalar()
    
    # 平均结果数
    avg_result_count = await self.db.query(
        func.avg(SearchHistory.result_count)
    ).filter(
        SearchHistory.created_at >= cutoff_date
    ).scalar()
    
    # 热门查询
    popular_queries = await self.db.query(
        SearchHistory.query_hash,
        SearchHistory.query,
        func.count(SearchHistory.id).label('count')
    ).filter(
        SearchHistory.created_at >= cutoff_date
    ).group_by(
        SearchHistory.query_hash,
        SearchHistory.query
    ).order_by(
        desc('count')
    ).limit(10).all()
    
    return {
        "period_days": days,
        "total_searches": total_searches,
        "avg_search_time_ms": round(avg_search_time or 0, 2),
        "avg_result_count": round(avg_result_count or 0, 2),
        "popular_queries": [
            {"query": q.query, "count": q.count}
            for q in popular_queries
        ]
    }
```
