## 1. 组件架构

### 1.1 组件树结构

```
Search.vue (主视图)
├── SearchInput.vue           # 搜索输入框
│   └── 多行文本输入
│   └── 快捷键支持 (Ctrl+Enter)
├── SearchConfig.vue          # 搜索配置面板
│   ├── 目标索引选择 (多选)
│   ├── Top-K 设置 (滑块+数字)
│   ├── 相似度阈值 (滑块+数字)
│   └── 度量类型展示 (只读)
├── SearchResults.vue         # 搜索结果展示
│   ├── ResultSummary         # 结果摘要
│   └── ResultCard.vue        # 结果卡片
│       └── ResultDetail      # 结果详情
└── SearchHistory.vue         # 搜索历史
    └── HistoryItem           # 历史项
```

### 1.2 组件职责分配

| 组件 | 职责 | 状态来源 |
|------|------|----------|
| Search.vue | 页面布局、组件协调 | searchStore |
| SearchInput | 接收用户输入 | props + emit |
| SearchConfig | 配置搜索参数 | props + emit |
| SearchResults | 展示搜索结果 | searchStore |
| SearchHistory | 管理历史记录 | searchStore |

  <div class="search-input">
    <t-textarea
      v-model="localValue"

      :autosize="{ minRows: 2, maxRows: 6 }"
      :maxlength="2000"
      @keydown="handleKeydown"

      <span class="char-count">{{ localValue.length }}/2000</span>

import { ref, watch } from 'vue'
import { SearchIcon } from 'tdesign-icons-vue-next'

const props = defineProps({
  modelValue: {
    type: String,
    default: ''
  },
  loading: {
    type: Boolean,
    default: false
  }
})

const emit = defineEmits(['update:modelValue', 'search'])

const localValue = ref(props.modelValue)

watch(() => props.modelValue, (val) => {
  localValue.value = val
})

watch(localValue, (val) => {
  emit('update:modelValue', val)
})

function handleKeydown(e) {
  // Ctrl+Enter 触发搜索
  if (e.ctrlKey && e.key === 'Enter') {
    e.preventDefault()
    handleSearch()
  }
}

  if (localValue.value.trim()) {
    emit('search')
  }
}
</script>

      />

    </div>
    
    <!-- 度量类型（只读显示） -->
    <div class="config-section">
      <label class="config-label">度量类型</label>
      <div class="metric-type-display">
        <t-tag 
          v-if="displayMetricType" 
          :theme="metricTypeTheme"
          variant="light"
        >
          {{ metricTypeLabel }}
        </t-tag>
        <span v-else class="metric-type-hint">选择索引后自动显示</span>
      </div>
      <p class="config-hint">度量类型由索引创建时确定</p>
    </div>

    <div class="config-actions">
      <t-button
        variant="outline"

        block

      >
        重置为默认
      </t-button>
    </div>
  </div>

<script setup>
import { ref, watch, computed } from 'vue'

const props = defineProps({
  config: Object,
  availableIndexes: Array,
  isLoadingIndexes: Boolean
})

const emit = defineEmits(['update:config', 'reset'])

const localConfig = ref({ ...props.config })

watch(() => props.config, (newVal) => {
  localConfig.value = { ...newVal }
}, { deep: true })

const indexOptions = computed(() => {
  return props.availableIndexes.map(idx => ({
    label: `${idx.name} (${idx.vector_count} 向量)`,
    value: idx.id
  }))
})

// 度量类型相关计算
const selectedIndexMetricTypes = computed(() => {
  if (localConfig.value.selectedIndexIds.length === 0) {
    if (props.availableIndexes.length === 0) return []
    const types = [...new Set(props.availableIndexes.map(idx => idx.metric_type || 'cosine'))]
    return types
  }
  
  const types = props.availableIndexes
    .filter(idx => localConfig.value.selectedIndexIds.includes(idx.id))
    .map(idx => idx.metric_type || 'cosine')
  
  return [...new Set(types)]
})

const displayMetricType = computed(() => {
  const types = selectedIndexMetricTypes.value
  if (types.length === 0) return null
  if (types.length === 1) return types[0]
  return 'mixed'
})

const metricTypeMap = {
  cosine: { label: '余弦相似度 (Cosine)', theme: 'primary' },
  euclidean: { label: '欧氏距离 (Euclidean)', theme: 'warning' },
  dot_product: { label: '点积 (Dot Product)', theme: 'success' }
}

const metricTypeLabel = computed(() => {
  const type = displayMetricType.value
  if (!type) return ''
  if (type === 'mixed') return '混合（多种度量类型）'
  return metricTypeMap[type]?.label || type
})

const metricTypeTheme = computed(() => {
  const type = displayMetricType.value
  if (!type || type === 'mixed') return 'default'
  return metricTypeMap[type]?.theme || 'default'
})

function handleIndexChange() {
  const types = selectedIndexMetricTypes.value
  if (types.length === 1) {
    localConfig.value.metricType = types[0]
  }
  handleConfigChange()
}

function handleConfigChange() {
  emit('update:config', { ...localConfig.value })
}

function handleReset() {
  localConfig.value = {
    selectedIndexIds: [],
    topK: 10,
    threshold: 0.5,
    metricType: 'cosine'
  }
  emit('reset')
  emit('update:config', { ...localConfig.value })
}
</script>

function formatTime(dateStr) {
  if (!dateStr) return ''
  
  let date
  if (typeof dateStr === 'string') {
    if (!dateStr.endsWith('Z') && !dateStr.includes('+') && !dateStr.includes('-', 10)) {
      date = new Date(dateStr + 'Z')
    } else {
      date = new Date(dateStr)
    }
  } else {
    date = new Date(dateStr)
  }
  
  const now = new Date()
  const diff = now - date
  
  if (diff < 60000) return '刚刚'
  if (diff < 3600000) return `${Math.floor(diff / 60000)}分钟前`
  if (diff < 86400000) return `${Math.floor(diff / 3600000)}小时前`
  if (diff < 604800000) return `${Math.floor(diff / 86400000)}天前`
  
  return date.toLocaleDateString()
}

  emit('select', item)
}

function handleDelete(id) {
  emit('delete', id)
}

  emit('clear')
}

  displayCount.value += 10
}

## 3. 状态管理

### 3.1 searchStore.js

```javascript
// stores/searchStore.js
import { defineStore } from 'pinia'
import { ref, computed } from 'vue'
import { searchApi } from '@/services/searchApi'

export const useSearchStore = defineStore('search', () => {
  // 配置状态
  const config = ref({
    selectedIndexIds: [],
    topK: 10,
    threshold: 0.5,
    metricType: 'cosine'
  })
  
  // 数据状态
  const availableIndexes = ref([])
  const results = ref([])
  const history = ref([])
  
  // 搜索状态
  const isSearching = ref(false)
  const searchTime = ref(0)
  const resultTotal = ref(0)
  const lastQuery = ref('')
  
  // 加载状态
  const loadingIndexes = ref(false)
  const loadingHistory = ref(false)
  
  // 计算属性
  const hasResults = computed(() => results.value.length > 0)
  
  // Actions
  async function fetchAvailableIndexes() {
    loadingIndexes.value = true
    try {
      const response = await searchApi.getIndexes()
      availableIndexes.value = response.data.indexes
    } catch (error) {
      console.error('Failed to fetch indexes:', error)
    } finally {
      loadingIndexes.value = false
    }
  }
  
  async function fetchHistory(limit = 50) {
    loadingHistory.value = true
    try {
      const response = await searchApi.getHistory(1, limit)
      history.value = response.data.items
    } catch (error) {
      console.error('Failed to fetch history:', error)
    } finally {
      loadingHistory.value = false
    }
  }
  
  async function search(query) {
    if (!query.trim()) return
    
    isSearching.value = true
    lastQuery.value = query
    
    try {
      const response = await searchApi.search({
        query,
        index_ids: config.value.selectedIndexIds,
        top_k: config.value.topK,
        threshold: config.value.threshold,
        metric_type: config.value.metricType
      })
      
      results.value = response.data.results
      searchTime.value = response.data.search_time_ms
      resultTotal.value = response.data.total
      
      // 刷新历史
      await fetchHistory()
      
      return response.data
    } catch (error) {
      console.error('Search failed:', error)
      results.value = []
      throw error
    } finally {
      isSearching.value = false
    }
  }
  
  async function deleteHistory(id) {
    try {
      await searchApi.deleteHistory(id)
      history.value = history.value.filter(h => h.id !== id)
    } catch (error) {
      console.error('Failed to delete history:', error)
      throw error
    }
  }
  
  async function clearHistory() {
    try {
      await searchApi.clearHistory()
      history.value = []
    } catch (error) {
      console.error('Failed to clear history:', error)
      throw error
    }
  }
  
  function updateConfig(newConfig) {
    config.value = { ...config.value, ...newConfig }
  }
  
  function resetConfig() {
    config.value = {
      selectedIndexIds: [],
      topK: 10,
      threshold: 0.5,
      metricType: 'cosine'
    }
  }
  
  function clearResults() {
    results.value = []
    searchTime.value = 0
    resultTotal.value = 0
    lastQuery.value = ''
  }
  
  return {
    // 状态
    config,
    availableIndexes,
    results,
    history,
    isSearching,
    searchTime,
    resultTotal,
    lastQuery,
    loadingIndexes,
    loadingHistory,
    hasResults,
    
    // Actions
    fetchAvailableIndexes,
    fetchHistory,
    search,
    deleteHistory,
    clearHistory,
    updateConfig,
    resetConfig,
    clearResults
  }
})
```

## 5. 工作流程

### 5.1 搜索执行流程

```
1. 用户输入查询文本
   └─> SearchInput v-model
       └─> Search.vue queryText

2. 用户按下搜索按钮或 Ctrl+Enter
   └─> SearchInput emit('search')
       └─> Search.vue handleSearch()
           └─> store.search(queryText)

3. 执行搜索
   └─> searchStore.search()
       ├─> 设置 isSearching = true
       ├─> 调用 searchApi.search()
       ├─> 更新 results, searchTime, resultTotal
       ├─> 刷新 history
       └─> 设置 isSearching = false

4. 展示结果
   └─> SearchResults 监听 store.results
       └─> 渲染 ResultCard 列表
```

## 6. 最佳实践

### 6.1 性能优化

```vue
<!-- 使用 v-memo 优化列表渲染 -->
<ResultCard
  v-for="result in results"
  :key="result.id"
  v-memo="[result.id, result.score]"
  :result="result"
/>
```

### 6.2 错误处理

```javascript
async function handleSearch() {
  try {
    await store.search(queryText.value)
  } catch (error) {
    message.error(`搜索失败: ${error.message}`)
  }
}
```

### 6.3 防抖搜索

```javascript
import { useDebounceFn } from '@vueuse/core'

const debouncedSearch = useDebounceFn(async () => {
  if (queryText.value.trim()) {
    await store.search(queryText.value)
  }
}, 300)
```
