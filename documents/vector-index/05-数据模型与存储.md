## 1. 数据模型概述

### 1.1 存储架构

向量索引模块采用**三层存储**架构：

| 存储层 | 用途 | 数据内容 |
|--------|------|----------|
| 关系数据库 | 元数据管理 | 索引信息、统计、操作日志、推荐日志 |
| 向量数据库 (Milvus) | 向量存储 | 稠密向量 + 稀疏向量（BM25）+ 元数据 |
| 文件系统 | 数据持久化 | BM25 统计数据、混合检索结果、索引元数据 |

### 1.2 设计原则

1. **Dify 方案分层**: 逻辑知识库与物理 Collection 分离，按维度拆分
2. **Milvus 优先**: 专注 Milvus 作为向量存储后端
3. **混合检索**: 稠密+稀疏双路向量支持
4. **查询优化**: Partition Key (doc_id) + 合理的索引设计

> **查询相关数据模型**: 查询历史（query_history）和混合检索结果持久化已移至 [检索查询模块文档](../search-query/03-前端组件与数据模型.md#2-数据模型)。

```python
# backend/src/models/vector_index.py

from sqlalchemy import Column, String, Integer, DateTime, Enum, ForeignKey, JSON, Float, Text, Boolean, func
from sqlalchemy.orm import relationship
from datetime import datetime
import enum

class IndexProvider(str, Enum):
    """向量数据库类型"""
    MILVUS = "MILVUS"

class IndexStatus(str, Enum):
    """索引状态"""
    BUILDING = "BUILDING"   # 构建中
    READY = "READY"         # 就绪
    UPDATING = "UPDATING"   # 更新中
    ERROR = "ERROR"         # 错误

class IndexType(str, Enum):
    """索引算法类型"""
    FLAT = "FLAT"
    IVF_FLAT = "IVF_FLAT"
    IVF_SQ8 = "IVF_SQ8"
    IVF_PQ = "IVF_PQ"
    HNSW = "HNSW"

class MetricType(str, Enum):
    """度量类型"""
    COSINE = "cosine"
    EUCLIDEAN = "euclidean"
    DOT_PRODUCT = "dot_product"

class VectorIndex(Base):
    __tablename__ = "vector_indexes"
    
    # 主键
    id = Column(Integer, primary_key=True, autoincrement=True)
    uuid = Column(String(36), nullable=False, unique=True)  # 外部UUID
    
    # 基本信息
    index_name = Column(String(255), nullable=False, unique=True)
    index_type = Column(Enum(IndexProvider), nullable=False, default=IndexProvider.MILVUS)
    algorithm_type = Column(String(50), default="FLAT")  # FLAT/IVF_FLAT/IVF_SQ8/IVF_PQ/HNSW
    dimension = Column(Integer, nullable=False)
    metric_type = Column(String(50), default="cosine")
    description = Column(Text, nullable=True)
    
    # 数据源关联
    embedding_result_id = Column(String(36), nullable=True)
    source_document_name = Column(String(255), nullable=True)
    source_model = Column(String(50), nullable=True)
    
    # Collection 管理（Dify 方案）
    collection_name = Column(String(255), default="default_collection")   # 逻辑知识库
    physical_collection_name = Column(String(255), nullable=True)  # 物理 Collection（如 default_collection_dim1024）
    
    # Provider 相关
    provider_index_id = Column(String(255), nullable=True)
    namespace = Column(String(255), default="default")
    index_params = Column(JSON, default={})
    file_path = Column(String(512), nullable=True)
    
    # 统计
    vector_count = Column(Integer, default=0)
    has_sparse = Column(Boolean, default=False)  # 是否包含BM25稀疏向量
    
    # 状态
    status = Column(Enum(IndexStatus), default=IndexStatus.BUILDING)
    error_message = Column(Text, nullable=True)
    
    # 时间戳
    created_at = Column(DateTime, server_default=func.now())
    updated_at = Column(DateTime, server_default=func.now(), onupdate=func.now())
    
    # 关系
    embedding_result = relationship("EmbeddingResult", back_populates="vector_indexes")
    document = relationship("Document", back_populates="vector_indexes")
    statistics = relationship("IndexStatistics", back_populates="vector_index", uselist=False)
    query_history = relationship("QueryHistory", back_populates="vector_index")  # → 详见检索查询模块
    operation_logs = relationship("IndexOperationLog", back_populates="vector_index")
    
    def __repr__(self):
        return f"<VectorIndex(id={self.id}, name={self.index_name}, provider={self.index_type})>"
```

**状态转换图**:
```
BUILDING → READY (索引构建成功)
BUILDING → ERROR (索引构建失败)
READY → UPDATING (增量更新)
UPDATING → READY (更新完成)
UPDATING → ERROR (更新失败)
ERROR → BUILDING (重新构建)
```

## 6. 持久化与恢复

### 6.1 FAISS 索引持久化

```python
import faiss
import numpy as np
import json

class FAISSPersistence:
    def __init__(self, base_path: str):
        self.base_path = base_path
    
    def save_index(self, index_id: str, index: faiss.Index, metadata: dict):
        """保存索引到文件"""
        # 保存索引文件
        index_path = f"{self.base_path}/{index_id}.faiss"
        faiss.write_index(index, index_path)
        
        # 保存元数据
        meta_path = f"{self.base_path}/{index_id}_meta.json"
        with open(meta_path, 'w') as f:
            json.dump(metadata, f, ensure_ascii=False, indent=2)
        
        return index_path
    
    def load_index(self, index_id: str) -> tuple:
        """从文件加载索引"""
        # 加载索引
        index_path = f"{self.base_path}/{index_id}.faiss"
        index = faiss.read_index(index_path)
        
        # 加载元数据
        meta_path = f"{self.base_path}/{index_id}_meta.json"
        with open(meta_path, 'r') as f:
            metadata = json.load(f)
        
        return index, metadata
    
    def delete_index(self, index_id: str):
        """删除索引文件"""
        import os
        
        index_path = f"{self.base_path}/{index_id}.faiss"
        meta_path = f"{self.base_path}/{index_id}_meta.json"
        
        if os.path.exists(index_path):
            os.remove(index_path)
        if os.path.exists(meta_path):
            os.remove(meta_path)
```

### 6.2 索引恢复流程

```python
async def recover_indexes(self):
    """恢复所有FAISS索引"""
    # 查询所有FAISS索引记录
    faiss_indexes = await self.db.query(VectorIndex).filter(
        VectorIndex.provider == IndexProvider.FAISS,
        VectorIndex.status == IndexStatus.READY
    ).all()
    
    recovered = 0
    failed = 0
    
    for idx_record in faiss_indexes:
        try:
            # 检查索引文件是否存在
            if not os.path.exists(idx_record.index_file):
                idx_record.status = IndexStatus.FAILED
                idx_record.error_message = "Index file not found"
                failed += 1
                continue
            
            # 加载索引到内存
            index, metadata = self.persistence.load_index(idx_record.id)
            self.loaded_indexes[idx_record.id] = index
            recovered += 1
            
        except Exception as e:
            idx_record.status = IndexStatus.FAILED
            idx_record.error_message = str(e)
            failed += 1
    
    await self.db.commit()
    return {"recovered": recovered, "failed": failed}
```

### 6.3 数据一致性保障

```python
async def create_index_with_transaction(self, config: IndexConfig):
    """事务性创建索引"""
    async with self.db.begin():
        # 1. 创建数据库记录
        index_record = VectorIndex(
            id=generate_uuid(),
            status=IndexStatus.BUILDING,
            **config.dict()
        )
        self.db.add(index_record)
        await self.db.flush()
        
        try:
            # 2. 在向量数据库中创建索引
            await self.provider.create_index(index_record.id, config)
            
            # 3. 添加向量数据
            await self.provider.add_vectors(
                index_record.id, 
                config.vectors,
                config.metadata
            )
            
            # 4. 持久化（FAISS）
            if config.provider == 'faiss':
                index_file = await self.persistence.save_index(
                    index_record.id,
                    self.provider.get_index(index_record.id),
                    config.metadata
                )
                index_record.index_file = index_file
            
            # 5. 更新状态
            index_record.status = IndexStatus.READY
            index_record.vector_count = len(config.vectors)
            
            await self.db.commit()
            return index_record
            
        except Exception as e:
            # 回滚：删除已创建的向量数据库索引
            await self.provider.delete_index(index_record.id)
            raise e
```

    vector_count INTEGER DEFAULT 0,
    total_queries INTEGER DEFAULT 0,
    avg_search_time_ms FLOAT DEFAULT 0,
    last_updated TIMESTAMP,
    FOREIGN KEY (index_id) REFERENCES vector_indexes(id)

CREATE TABLE IF NOT EXISTS query_history (

    top_k INTEGER NOT NULL,
    filters TEXT,
    results_count INTEGER DEFAULT 0,
    search_time_ms FLOAT NOT NULL,
    search_mode VARCHAR(20),
    response_time_ms FLOAT,
    created_at TIMESTAMP DEFAULT CURRENT_TIMESTAMP,
    FOREIGN KEY (index_id) REFERENCES vector_indexes(id)
);

CREATE TABLE IF NOT EXISTS index_operation_logs (

    error_message TEXT,
    created_at TIMESTAMP DEFAULT CURRENT_TIMESTAMP
