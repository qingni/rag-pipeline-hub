## 1. 数据模型概述

### 1.1 存储架构

向量索引模块采用**三层存储**架构：

| 存储层 | 用途 | 数据内容 |
|--------|------|----------|
| 关系数据库 | 元数据管理 | 索引信息、统计、历史 |
| 向量数据库 | 向量存储 | 向量数据、元数据 |
| 文件系统 | 索引持久化 | FAISS索引文件 |

### 1.2 设计原则

1. **元数据与向量分离**: 元数据存关系库，向量存专用库
2. **多Provider支持**: 统一接口，支持多种向量数据库
3. **持久化保障**: FAISS索引支持文件持久化
4. **查询优化**: 合理的索引设计

## 3. 数据库模型详解

### 3.1 VectorIndex 模型

```python
# backend/src/models/vector_index.py

from sqlalchemy import Column, String, Integer, DateTime, Enum, ForeignKey, JSON, Float
from sqlalchemy.orm import relationship
from datetime import datetime
import enum

class IndexStatus(enum.Enum):
    """索引状态枚举"""
    PENDING = "pending"       # 等待创建
    BUILDING = "building"     # 构建中
    READY = "ready"           # 就绪
    FAILED = "failed"         # 失败
    DELETED = "deleted"       # 已删除

class IndexProvider(enum.Enum):
    """向量数据库类型"""
    MILVUS = "milvus"
    FAISS = "faiss"

class IndexAlgorithm(enum.Enum):
    """索引算法类型"""
    FLAT = "FLAT"
    HNSW = "HNSW"
    IVF_FLAT = "IVF_FLAT"
    IVF_PQ = "IVF_PQ"

class MetricType(enum.Enum):
    """度量类型"""
    COSINE = "cosine"
    EUCLIDEAN = "euclidean"
    DOT_PRODUCT = "dot_product"

class VectorIndex(Base):
    """向量索引模型"""
    __tablename__ = "vector_indexes"
    
    # 主键
    id = Column(String(36), primary_key=True)
    
    # 关联
    embedding_result_id = Column(String(36), ForeignKey("embedding_results.id"), nullable=True)
    document_id = Column(String(36), ForeignKey("documents.id"), nullable=True)
    
    # 基本信息
    index_name = Column(String(200), nullable=False, comment="索引名称")
    
    # 索引配置
    provider = Column(Enum(IndexProvider), nullable=False, comment="向量数据库类型")
    algorithm = Column(Enum(IndexAlgorithm), nullable=False, comment="索引算法")
    algorithm_params = Column(JSON, nullable=True, comment="算法参数")
    metric_type = Column(Enum(MetricType), nullable=False, comment="度量类型")
    
    # 向量信息
    dimension = Column(Integer, nullable=False, comment="向量维度")
    vector_count = Column(Integer, default=0, comment="向量数量")
    
    # 状态
    status = Column(Enum(IndexStatus), default=IndexStatus.PENDING, comment="状态")
    error_message = Column(String(1000), nullable=True, comment="错误信息")
    
    # 存储信息
    index_file = Column(String(500), nullable=True, comment="索引文件路径(FAISS)")
    collection_name = Column(String(200), nullable=True, comment="Collection名称(Milvus)")
    
    # 时间戳
    created_at = Column(DateTime, default=datetime.utcnow, comment="创建时间")
    updated_at = Column(DateTime, default=datetime.utcnow, onupdate=datetime.utcnow, comment="更新时间")
    
    # 关系
    embedding_result = relationship("EmbeddingResult", back_populates="vector_indexes")
    document = relationship("Document", back_populates="vector_indexes")
    statistics = relationship("IndexStatistics", back_populates="vector_index", uselist=False)
    query_history = relationship("QueryHistory", back_populates="vector_index")
    
    def __repr__(self):
        return f"<VectorIndex(id={self.id}, name={self.index_name}, provider={self.provider})>"
```

## 6. 持久化与恢复

### 6.1 FAISS 索引持久化

```python
import faiss
import numpy as np
import json

class FAISSPersistence:
    def __init__(self, base_path: str):
        self.base_path = base_path
    
    def save_index(self, index_id: str, index: faiss.Index, metadata: dict):
        """保存索引到文件"""
        # 保存索引文件
        index_path = f"{self.base_path}/{index_id}.faiss"
        faiss.write_index(index, index_path)
        
        # 保存元数据
        meta_path = f"{self.base_path}/{index_id}_meta.json"
        with open(meta_path, 'w') as f:
            json.dump(metadata, f, ensure_ascii=False, indent=2)
        
        return index_path
    
    def load_index(self, index_id: str) -> tuple:
        """从文件加载索引"""
        # 加载索引
        index_path = f"{self.base_path}/{index_id}.faiss"
        index = faiss.read_index(index_path)
        
        # 加载元数据
        meta_path = f"{self.base_path}/{index_id}_meta.json"
        with open(meta_path, 'r') as f:
            metadata = json.load(f)
        
        return index, metadata
    
    def delete_index(self, index_id: str):
        """删除索引文件"""
        import os
        
        index_path = f"{self.base_path}/{index_id}.faiss"
        meta_path = f"{self.base_path}/{index_id}_meta.json"
        
        if os.path.exists(index_path):
            os.remove(index_path)
        if os.path.exists(meta_path):
            os.remove(meta_path)
```

### 6.2 索引恢复流程

```python
async def recover_indexes(self):
    """恢复所有FAISS索引"""
    # 查询所有FAISS索引记录
    faiss_indexes = await self.db.query(VectorIndex).filter(
        VectorIndex.provider == IndexProvider.FAISS,
        VectorIndex.status == IndexStatus.READY
    ).all()
    
    recovered = 0
    failed = 0
    
    for idx_record in faiss_indexes:
        try:
            # 检查索引文件是否存在
            if not os.path.exists(idx_record.index_file):
                idx_record.status = IndexStatus.FAILED
                idx_record.error_message = "Index file not found"
                failed += 1
                continue
            
            # 加载索引到内存
            index, metadata = self.persistence.load_index(idx_record.id)
            self.loaded_indexes[idx_record.id] = index
            recovered += 1
            
        except Exception as e:
            idx_record.status = IndexStatus.FAILED
            idx_record.error_message = str(e)
            failed += 1
    
    await self.db.commit()
    return {"recovered": recovered, "failed": failed}
```

### 6.3 数据一致性保障

```python
async def create_index_with_transaction(self, config: IndexConfig):
    """事务性创建索引"""
    async with self.db.begin():
        # 1. 创建数据库记录
        index_record = VectorIndex(
            id=generate_uuid(),
            status=IndexStatus.BUILDING,
            **config.dict()
        )
        self.db.add(index_record)
        await self.db.flush()
        
        try:
            # 2. 在向量数据库中创建索引
            await self.provider.create_index(index_record.id, config)
            
            # 3. 添加向量数据
            await self.provider.add_vectors(
                index_record.id, 
                config.vectors,
                config.metadata
            )
            
            # 4. 持久化（FAISS）
            if config.provider == 'faiss':
                index_file = await self.persistence.save_index(
                    index_record.id,
                    self.provider.get_index(index_record.id),
                    config.metadata
                )
                index_record.index_file = index_file
            
            # 5. 更新状态
            index_record.status = IndexStatus.READY
            index_record.vector_count = len(config.vectors)
            
            await self.db.commit()
            return index_record
            
        except Exception as e:
            # 回滚：删除已创建的向量数据库索引
            await self.provider.delete_index(index_record.id)
            raise e
```
