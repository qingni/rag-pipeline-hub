## 1. 组件架构

### 1.1 组件树结构

```
VectorIndex.vue (主视图)
├── EmbeddingTaskSelector.vue  # 向量化结果选择器
├── ProviderSelector.vue       # 向量数据库选择器（当前仅 Milvus）
├── AlgorithmSelector.vue      # 索引算法选择器（FLAT/IVF_FLAT/IVF_SQ8/IVF_PQ/HNSW）
├── MetricSelector.vue         # 度量类型选择器
├── RecommendPanel.vue         # 智能推荐面板（新增）
├── IndexCreate.vue            # 索引创建面板
│   ├── MatchedIndexDisplay    # 匹配索引展示
│   ├── SparseToggle           # BM25 稀疏向量开关（新增）
│   └── CreateButton           # 创建按钮
├── IndexList.vue              # 索引列表
│   └── IndexCard.vue          # 索引卡片
└── CollectionList.vue         # Collection 列表（新增）
```

> **检索组件**: `HybridSearch.vue`、`VectorSearch.vue` 等检索相关前端组件已移至 [检索查询模块文档](../search-query/03-前端组件与数据模型.md)。

### 1.2 组件职责分配

| 组件 | 职责 | 状态来源 |
|------|------|----------|
| VectorIndex.vue | 页面布局、组件协调 | vectorIndexStore |

| MetricSelector | 选择度量类型 | props + emit |

| IndexList | 展示索引列表 | vectorIndexStore |
| CollectionList | 展示 Collection 列表（Dify 方案） | vectorIndexStore |

> **检索组件**: HybridSearch、VectorSearch 等检索相关组件的详细说明见 [检索查询模块文档](../search-query/03-前端组件与数据模型.md)。

## 2. 核心组件详解

### 2.1 VectorIndex.vue (主视图)

```vue
<template>
  <div class="vector-index-page">
    <div class="page-header">
      <h1>向量索引</h1>
      <p>将向量化结果导入向量数据库，建立高效索引</p>
    </div>
    
    <div class="page-content">
      <!-- 左侧配置区 -->
      <div class="config-panel">
        <EmbeddingTaskSelector
          v-model="selectedEmbeddingResult"
          :embedding-results="embeddingResults"
          :loading="loadingEmbeddings"
        />
        
        <ProviderSelector
          v-model="selectedProvider"
          :providers="providers"
        />
        
        <AlgorithmSelector
          v-model="selectedAlgorithm"
          :algorithms="availableAlgorithms"
          :provider="selectedProvider"
        />
        
        <MetricSelector
          v-model="selectedMetricType"
          :metric-types="metricTypes"
        />
        
        <IndexCreate
          :embedding-result="selectedEmbeddingResult"
          :provider="selectedProvider"
          :algorithm="selectedAlgorithm"
          :metric-type="selectedMetricType"
          :matched-index="matchedIndex"
          @create="handleCreate"
        />
      </div>
      
      <!-- 右侧列表区 -->
      <div class="list-panel">
        <IndexList
          :indexes="indexes"
          :loading="loadingIndexes"
          @delete="handleDelete"
          @view="handleView"
        />
      </div>
    </div>
  </div>
</template>

<script setup>
import { ref, computed, watch, onMounted } from 'vue'
import { useVectorIndexStore } from '@/stores/vectorIndexStore'

const store = useVectorIndexStore()

// 选择状态
const selectedEmbeddingResult = ref(null)
const selectedProvider = ref('milvus')
const selectedAlgorithm = ref('HNSW')
const selectedMetricType = ref('cosine')

// 计算属性
const embeddingResults = computed(() => store.embeddingResults)
const providers = computed(() => store.providers)
const metricTypes = computed(() => store.metricTypes)
const indexes = computed(() => store.indexes)
const matchedIndex = computed(() => store.matchedIndex)

const availableAlgorithms = computed(() => {
  const provider = providers.value.find(p => p.name === selectedProvider.value)
  return provider?.supported_algorithms || []
})

// 监听配置变化，检查匹配索引
watch(
  [selectedEmbeddingResult, selectedProvider, selectedAlgorithm, selectedMetricType],
  async ([embResult, provider, algorithm, metric]) => {
    if (embResult && provider && algorithm && metric) {
      await store.checkMatchedIndex({
        embeddingResultId: embResult.id,
        provider,
        algorithm,
        metricType: metric
      })
    }
  }
)

// 生命周期
onMounted(async () => {
  await Promise.all([
    store.fetchEmbeddingResults(),
    store.fetchProviders(),
    store.fetchIndexes()
  ])
})

// 方法
async function handleCreate(overwrite = false) {
  await store.createIndex({
    embeddingResultId: selectedEmbeddingResult.value.id,
    provider: selectedProvider.value,
    algorithm: selectedAlgorithm.value,
    metricType: selectedMetricType.value,
    overwrite
  })
}

async function handleDelete(indexId) {
  await store.deleteIndex(indexId)
}

function handleView(index) {
  // 查看索引详情
}
</script>
```

<template>

})

  return props.providers.find(p => p.name === selected.value)

  emit('update:modelValue', val)

  return name === 'milvus' ? DatabaseIcon : HardDriveIcon

  emit('update:modelValue', val)

### 2.3 AlgorithmSelector.vue

```vue
<template>
  <div class="algorithm-selector">
    <h3>索引算法</h3>
    
    <t-select
      v-model="selected"
      :options="algorithmOptions"
      placeholder="选择索引算法"
      @change="handleChange"
    />
    
    <div v-if="currentAlgorithm" class="algorithm-info">
      <p>{{ currentAlgorithm.description }}</p>
      
      <!-- 算法参数配置 -->
      <div v-if="currentAlgorithm.params?.length" class="algorithm-params">
        <h4>算法参数</h4>
        <div v-for="param in currentAlgorithm.params" :key="param.name" class="param-item">
          <label>{{ param.name }}</label>
          <t-input-number
            v-model="params[param.name]"
            :min="param.min"
            :max="param.max"
            :step="param.step || 1"
          />
          <span class="param-hint">{{ param.description }}</span>
        </div>
      </div>
    </div>
  </div>
</template>

<script setup>
import { ref, computed, watch, reactive } from 'vue'

const props = defineProps({
  modelValue: String,
  algorithms: Array,
  provider: String
})

const emit = defineEmits(['update:modelValue', 'update:params'])

const selected = ref(props.modelValue)
const params = reactive({})

const algorithmOptions = computed(() => {
  return props.algorithms.map(alg => ({
    value: alg.name,
    label: alg.display_name
  }))
})

const currentAlgorithm = computed(() => {
  return props.algorithms.find(a => a.name === selected.value)
})

// 初始化参数默认值
watch(currentAlgorithm, (alg) => {
  if (alg?.params) {
    alg.params.forEach(p => {
      if (!(p.name in params)) {
        params[p.name] = p.default
      }
    })
  }
}, { immediate: true })

watch(selected, (val) => {
  emit('update:modelValue', val)
})

watch(params, (val) => {
  emit('update:params', { ...val })
}, { deep: true })
</script>
```

      </t-alert>

        </t-button>
      </div>
    </div>
    
    <!-- 创建按钮 -->
    <div v-else class="create-actions">
      <t-button
        theme="primary"
        :loading="isCreating"
        :disabled="!canCreate"
        @click="handleCreate"
      >
        {{ isCreating ? '创建中...' : '创建索引' }}
      </t-button>
    </div>
    
    <!-- 进度显示 -->
    <div v-if="isCreating" class="progress-section">
      <t-progress :percentage="progress" />
      <p class="progress-text">{{ progressText }}</p>
    </div>
  </div>
</template>

<script setup>
import { computed } from 'vue'
import { useVectorIndexStore } from '@/stores/vectorIndexStore'

const props = defineProps({
  embeddingResult: Object,
  provider: String,
  algorithm: String,
  metricType: String,
  matchedIndex: Object
})

const emit = defineEmits(['create', 'use-existing'])

const store = useVectorIndexStore()

const isCreating = computed(() => store.isCreating)
const progress = computed(() => store.createProgress)
const progressText = computed(() => store.createProgressText)

const metricTypeLabel = computed(() => {
  const labels = {
    cosine: '余弦相似度',
    euclidean: '欧氏距离',
    dot_product: '点积'
  }
  return labels[props.metricType] || props.metricType
})

const canCreate = computed(() => {
  return props.embeddingResult && 
         props.provider && 
         props.algorithm && 
         props.metricType &&
         !isCreating.value
})

function handleCreate() {
  emit('create', false)
}

function handleOverwrite() {
  emit('create', true)
}

function handleUseExisting() {
  emit('use-existing', props.matchedIndex)
}

function formatTime(dateStr) {
  return new Date(dateStr).toLocaleString()
}
</script>

## 3. 状态管理

### 3.1 vectorIndexStore.js

```javascript
// stores/vectorIndexStore.js
import { defineStore } from 'pinia'
import { ref, computed } from 'vue'
import { vectorIndexApi } from '@/services/vectorIndexApi'

export const useVectorIndexStore = defineStore('vectorIndex', () => {
  // 数据状态
  const embeddingResults = ref([])
  const providers = ref([])
  const algorithms = ref([])
  const metricTypes = ref([
    { name: 'cosine', label: '余弦相似度' },
    { name: 'euclidean', label: '欧氏距离' },
    { name: 'dot_product', label: '点积' }
  ])
  const indexes = ref([])
  const matchedIndex = ref(null)
  
  // 创建状态
  const isCreating = ref(false)
  const createProgress = ref(0)
  const createProgressText = ref('')
  
  // 加载状态
  const loadingEmbeddings = ref(false)
  const loadingIndexes = ref(false)
  
  // 计算属性
  const readyIndexes = computed(() => {
    return indexes.value.filter(idx => idx.status === 'ready')
  })
  
  // Actions
  async function fetchEmbeddingResults() {
    loadingEmbeddings.value = true
    try {
      const response = await vectorIndexApi.getEmbeddingResults()
      embeddingResults.value = response.data.items.filter(
        item => item.status === 'completed'
      )
    } catch (error) {
      console.error('Failed to fetch embedding results:', error)
    } finally {
      loadingEmbeddings.value = false
    }
  }
  
  async function fetchProviders() {
    try {
      const response = await vectorIndexApi.getProviders()
      providers.value = response.data.providers
    } catch (error) {
      console.error('Failed to fetch providers:', error)
    }
  }
  
  async function fetchIndexes() {
    loadingIndexes.value = true
    try {
      const response = await vectorIndexApi.getIndexes()
      indexes.value = response.data.items
    } catch (error) {
      console.error('Failed to fetch indexes:', error)
    } finally {
      loadingIndexes.value = false
    }
  }
  
  async function checkMatchedIndex(config) {
    try {
      const response = await vectorIndexApi.checkMatch(config)
      matchedIndex.value = response.data.matched ? response.data.index : null
    } catch (error) {
      console.error('Failed to check matched index:', error)
      matchedIndex.value = null
    }
  }
  
  async function createIndex(config) {
    isCreating.value = true
    createProgress.value = 0
    createProgressText.value = '准备创建索引...'
    
    try {
      createProgressText.value = '正在创建索引...'
      createProgress.value = 30
      
      const response = await vectorIndexApi.createFromEmbedding({
        embedding_result_id: config.embeddingResultId,
        provider: config.provider,
        algorithm: config.algorithm,
        metric_type: config.metricType,
        overwrite: config.overwrite
      })
      
      createProgress.value = 100
      createProgressText.value = '索引创建完成'
      
      // 刷新列表
      await fetchIndexes()
      matchedIndex.value = null
      
      return response.data
    } catch (error) {
      createProgressText.value = `创建失败: ${error.message}`
      throw error
    } finally {
      isCreating.value = false
    }
  }
  
  async function deleteIndex(indexId) {
    try {
      await vectorIndexApi.deleteIndex(indexId)
      indexes.value = indexes.value.filter(idx => idx.id !== indexId)
    } catch (error) {
      console.error('Failed to delete index:', error)
      throw error
    }
  }
  
  function reset() {
    matchedIndex.value = null
    isCreating.value = false
    createProgress.value = 0
    createProgressText.value = ''
  }
  
  return {
    // 状态
    embeddingResults,
    providers,
    algorithms,
    metricTypes,
    indexes,
    readyIndexes,
    matchedIndex,
    isCreating,
    createProgress,
    createProgressText,
    loadingEmbeddings,
    loadingIndexes,
    
    // Actions
    fetchEmbeddingResults,
    fetchProviders,
    fetchIndexes,
    checkMatchedIndex,
    createIndex,
    deleteIndex,
    reset
  }
})
```

## 5. 工作流程

### 5.1 索引创建流程

```
1. 用户选择向量化结果
   └─> EmbeddingTaskSelector emit
       └─> VectorIndex.vue 更新 selectedEmbeddingResult

2. 用户选择向量数据库
   └─> ProviderSelector emit
       └─> VectorIndex.vue 更新 selectedProvider

3. 用户选择索引算法
   └─> AlgorithmSelector emit
       └─> VectorIndex.vue 更新 selectedAlgorithm

4. 用户选择度量类型
   └─> MetricSelector emit
       └─> VectorIndex.vue 更新 selectedMetricType

5. 自动检查匹配索引
   └─> watch 触发 store.checkMatchedIndex()
       └─> 更新 matchedIndex

6. 用户点击创建
   └─> IndexCreate emit('create')
       └─> store.createIndex()
           ├─> 调用 API
           ├─> 更新进度
           └─> 刷新列表
```
