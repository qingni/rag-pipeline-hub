## 1. 数据模型概述

### 1.1 存储架构

文本生成模块主要存储生成历史记录：

| 存储类型 | 用途 | 数据内容 |
|---------|------|----------|
| 关系数据库 | 历史记录 | 生成历史、配置、结果 |
| 内存 | 运行时状态 | 活跃请求、流式控制 |

### 1.2 设计原则

1. **完整记录**: 保存完整的请求和响应信息
2. **软删除**: 支持单条删除与全部清空，统一采用软删除
3. **自动清理**: 超出限制时自动清理旧记录
4. **快速查询**: 支持按时间、状态快速查询

## 3. 数据库模型详解

### 3.1 GenerationHistory 模型

```python
# backend/src/models/generation.py

from datetime import datetime
from enum import Enum
from sqlalchemy import Boolean, Column, DateTime, Float, Integer, String, Text, JSON
from sqlalchemy.sql import func
from ..storage.database import Base


class GenerationStatus(str, Enum):
    """生成请求状态"""
    PENDING = "pending"
    GENERATING = "generating"
    COMPLETED = "completed"
    FAILED = "failed"
    CANCELLED = "cancelled"


class GenerationHistory(Base):
    """生成历史模型"""
    
    __tablename__ = "generation_history"
    
    # 主键
    id = Column(Integer, primary_key=True, autoincrement=True)
    
    # 请求标识
    request_id = Column(String(36), unique=True, nullable=False, index=True)
    
    # 问题和模型配置
    question = Column(Text, nullable=False)
    model = Column(String(50), nullable=False)
    temperature = Column(Float, default=0.7)
    max_tokens = Column(Integer, default=4096)
    
    # 上下文信息
    context_summary = Column(Text, nullable=True)  # 上下文摘要
    context_sources = Column(JSON, nullable=True)  # 完整上下文来源
    
    # 生成结果
    answer = Column(Text, nullable=True)
    token_usage = Column(JSON, nullable=True)
    processing_time_ms = Column(Float, nullable=True)
    
    # 状态
    status = Column(
        String(20), 
        nullable=False, 
        default=GenerationStatus.PENDING.value, 
        index=True
    )
    error_message = Column(Text, nullable=True)
    
    # 时间戳
    created_at = Column(DateTime, nullable=False, default=func.now(), index=True)
    updated_at = Column(DateTime, nullable=False, default=func.now(), onupdate=func.now())
    
    # 软删除
    is_deleted = Column(Boolean, default=False, index=True)
    
    def __repr__(self) -> str:
        return f"<GenerationHistory(id={self.id}, request_id={self.request_id}, status={self.status})>"
    
    def to_dict(self) -> dict:
        """转换为字典"""
        return {
            "id": self.id,
            "request_id": self.request_id,
            "question": self.question,
            "model": self.model,
            "temperature": self.temperature,
            "max_tokens": self.max_tokens,
            "context_summary": self.context_summary,
            "context_sources": self.context_sources,
            "answer": self.answer,
            "token_usage": self.token_usage,
            "processing_time_ms": self.processing_time_ms,
            "status": self.status,
            "error_message": self.error_message,
            "created_at": self.created_at.isoformat() if self.created_at else None,
            "updated_at": self.updated_at.isoformat() if self.updated_at else None,
            "is_deleted": self.is_deleted
        }
    
    def get_answer_preview(self, max_length: int = 200) -> str:
        """获取回答预览"""
        if not self.answer:
            return ""
        if len(self.answer) <= max_length:
            return self.answer
        return self.answer[:max_length] + "..."
```

### 5.2 查询历史列表

```python
async def list_history(
    page: int = 1,
    page_size: int = 20,
    status: Optional[str] = None,
    db: Session = Depends(get_db),
):
    """获取历史列表"""
    query = db.query(GenerationHistory).filter(
        GenerationHistory.is_deleted == False
    )
    
    if status:
        query = query.filter(GenerationHistory.status == status)
    
    total = query.count()
    total_pages = (total + page_size - 1) // page_size
    
    items = query.order_by(GenerationHistory.created_at.desc()).offset(
        (page - 1) * page_size
    ).limit(page_size).all()
    
    return {
        "items": [
            {
                "id": item.id,
                "request_id": item.request_id,
                "question": item.question,
                "answer_preview": item.get_answer_preview(),
                "model": item.model,
                "status": item.status,
                "created_at": item.created_at,
            }
            for item in items
        ],
        "total": total,
        "page": page,
        "page_size": page_size,
        "total_pages": total_pages,
    }
```

### 5.3 软删除

```python
async def delete_history(id: int, db: Session):
    """软删除历史记录"""
    history = db.query(GenerationHistory).filter(
        GenerationHistory.id == id,
        GenerationHistory.is_deleted == False,
    ).first()
    
    if not history:
        raise HTTPException(status_code=404, detail="History not found")
    
    history.is_deleted = True
    db.commit()
    
    return {"success": True, "message": "History deleted"}
```

### 5.4 清空全部历史

```python
async def clear_history(db: Session):
    """清空全部历史（软删除）"""
    db.query(GenerationHistory).filter(
        GenerationHistory.is_deleted == False
    ).update({"is_deleted": True})
    db.commit()

    return {"success": True, "message": "All history cleared"}
```

## 6. 数据清理策略

### 6.1 自动清理规则

| 规则 | 条件 | 动作 |
|------|------|------|
| 数量限制 | 超过100条 | 软删除最旧的记录 |
| 时间限制 | 当前实现未启用固定天数淘汰 | 如后续启用可按时间软删除 |

### 6.2 清理实现

```python
@staticmethod
def cleanup_old_history(db_session, max_records: int = 100) -> int:
    """清理超出限制的旧历史记录"""
    from sqlalchemy import func
    
    # 统计未删除的记录数
    count = db_session.query(func.count(GenerationHistory.id)).filter(
        GenerationHistory.is_deleted == False
    ).scalar()
    
    if count <= max_records:
        return 0
    
    # 获取需要删除的记录ID
    records_to_delete = count - max_records
    oldest_records = db_session.query(GenerationHistory.id).filter(
        GenerationHistory.is_deleted == False
    ).order_by(GenerationHistory.created_at.asc()).limit(records_to_delete).all()
    
    ids_to_delete = [r[0] for r in oldest_records]
    
    # 软删除
    db_session.query(GenerationHistory).filter(
        GenerationHistory.id.in_(ids_to_delete)
    ).update({"is_deleted": True}, synchronize_session=False)
    
    db_session.commit()
    
    return len(ids_to_delete)
```


### 6.4 当前保留策略说明

1. 流式生成在成功结束时写入完整历史记录。
2. 取消生成时，如果已有历史记录，会将状态更新为 `cancelled`。
3. 清空历史和单条删除都只修改 `is_deleted`，不做物理删除。
4. 当前实现主要依赖“数量限制 + 手动清空”，未启用固定保留天数任务。

## 8. 数据统计

### 8.1 统计查询示例

```python
async def get_generation_statistics(db: Session, days: int = 7) -> dict:
    """获取生成统计信息"""
    from datetime import datetime, timedelta
    from sqlalchemy import func
    
    cutoff_date = datetime.utcnow() - timedelta(days=days)
    
    # 总生成次数
    total_generations = db.query(func.count(GenerationHistory.id)).filter(
        GenerationHistory.created_at >= cutoff_date,
        GenerationHistory.is_deleted == False
    ).scalar()
    
    # 成功率
    completed_count = db.query(func.count(GenerationHistory.id)).filter(
        GenerationHistory.created_at >= cutoff_date,
        GenerationHistory.status == 'completed',
        GenerationHistory.is_deleted == False
    ).scalar()
    
    success_rate = completed_count / total_generations if total_generations > 0 else 0
    
    # 平均处理时间
    avg_processing_time = db.query(func.avg(GenerationHistory.processing_time_ms)).filter(
        GenerationHistory.created_at >= cutoff_date,
        GenerationHistory.status == 'completed',
        GenerationHistory.is_deleted == False
    ).scalar()
    
    # 模型使用分布
    model_distribution = db.query(
        GenerationHistory.model,
        func.count(GenerationHistory.id).label('count')
    ).filter(
        GenerationHistory.created_at >= cutoff_date,
        GenerationHistory.is_deleted == False
    ).group_by(GenerationHistory.model).all()
    
    return {
        "period_days": days,
        "total_generations": total_generations,
        "success_rate": round(success_rate * 100, 2),
        "avg_processing_time_ms": round(avg_processing_time or 0, 2),
        "model_distribution": [
            {"model": m.model, "count": m.count}
            for m in model_distribution
        ]
    }
```
